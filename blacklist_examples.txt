# Ejemplo de curl para agregar email a blacklist
curl -X POST http://localhost:5000/blacklists/ \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer fixedtoken" \
  -d '{"email": "test@example.com", "app_uuid": "123e4567-e89b-12d3-a456-426614174000", "blocked_reason": "spam"}'

# Ejemplo de curl para consultar email en blacklist
curl -X GET http://localhost:5000/blacklists/test@example.com \
  -H "Authorization: Bearer fixedtoken"
